09 — Handoff Tokens

코드로 바로 쓰는 토큰

Claude Code에 그대로 붙여넣을 수 있는 CSS 변수 세트입니다. Pretendard는 CDN으로 로드합니다.

/* Soez Design Tokens */
:root {
  --honey-100: #FDEFD3;  /* 꿀색 틴트·배경강조 */
  --honey-300: #FAC775;  /* 밝은 꿀색·배지 */
  --honey-500: #F4A93C;  /* 메인 꿀색·버튼·강조 */
  --honey-700: #C5851E;  /* 머스타드·링크 */
  --comb-700 : #8A5A2B;  /* 벌집 브라운·포인트 */

  --ink-900  : #2A2926;  /* 본문 진한 글자 */
  --ink-600  : #5C574E;
  --ink-400  : #8A867B;
  --line-200 : #E6E2D6;  /* 보더 */
  --surface  : #FFFFFF;
  --surface-2: #FBF6EA;
  --bg       : #FDF6E8;  /* 화면 전체 크림 배경 */

  --success  : #4F9A5E;  /* 출석 */
  --warning  : #E8804D;  /* 지각 */
  --danger   : #CE5246;  /* 결석 */
  --info     : #4E86A6;  /* 정보·보충 */

  --radius-sm: 10px;  --radius-md: 12px;
  --radius-lg: 18px;  --radius-xl: 24px;
  --shadow-card: 0 2px 18px rgba(138,90,43,0.06);
  --font-sans : "Pretendard", -apple-system, sans-serif;  /* UI·본문 전체 */
  --font-logo : "Jua", "Pretendard", sans-serif;  /* 로고 워드마크 전용 */
}
